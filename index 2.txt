  <h1>Calculadora de Volume</h1>

  <input id="aluno" placeholder="Aluno" />
  <input id="exercicio" placeholder="Exercício" />
  <input id="series" type="number" placeholder="Séries" />
  <input id="reps" type="number" placeholder="Reps" />
  <input id="carga" type="number" step="0.5" placeholder="Carga" />
  <button id="addBtn">Adicionar</button>

  <script>
    const WEB_APP_URL = "COLE_AQUI_SUA_URL_EXEC";

    async function enviar() {
      const aluno = document.getElementById("aluno").value.trim();
      const exercicio = document.getElementById("exercicio").value.trim();
      const series = document.getElementById("series").value;
      const reps = document.getElementById("reps").value;
      const carga = document.getElementById("carga").value;

      if (!aluno || !exercicio || !series || !reps || carga === "") {
        alert("Preencha todos os campos");
        return;
      }

      const dados = new URLSearchParams({
        aluno, exercicio, series, reps, carga
      });

      try {
        const resp = await fetch(WEB_APP_URL, {
          method: "POST",
          headers: { "Content-Type": "application/x-www-form-urlencoded;charset=UTF-8" },
          body: dados.toString()
        });

        const txt = await resp.text();
        let json = {};
        try { json = JSON.parse(txt); } catch(e) {}

        if (!resp.ok || !json.ok) {
          throw new Error(json.erro || ("HTTP " + resp.status));
        }

        alert("Enviado para planilha com sucesso");
      } catch (err) {
        console.error(err);
        alert("Erro ao enviar: " + err.message);
      }
    }

    document.getElementById("addBtn").addEventListener("click", enviar);
  </script>
